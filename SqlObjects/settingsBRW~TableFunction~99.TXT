create function dbo.settingsBRW(
 @parentName varchar(50)
) returns @rt table(
 id int,
 display varchar(max),
 rowType int,
 ord varchar(50),
 Description varchar(200),
 name varchar(max),
 value varchar(max)
) as
begin

 insert @rt
 select
  id * -1,
  dbo.splitF(settingName,'.',2) + ':' + upper(settingValue),
  0,
  settingName,
  Description,
  settingName,
  settingValue
 from settings
 where dbo.splitF(settingName,'.',1) = @parentName

 return
end
