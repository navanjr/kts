create proc diagFix_glDetailSyncDate(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on

 if @method = 'GET'
 begin  

  select
   @class = 'General',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(distinct slink),  
   @message = 'glDetail.date is out of Sync'
  from dbo.glDetailReports 
  where isnumeric(sourceDate+'.e0') = 1
   and sourceDate > '1'
   and cast(date as numeric(38,0)) != sourceDate

  exec dbo.diagnosticsAutoHelper @@procid, @blockAutoFix

  return
 end

 if @method = 'SHOW'
 begin  

  select distinct slink, DATE, sourceDate
  from dbo.glDetailReports 
  where isnumeric(sourceDate+'.e0') = 1
   and sourceDate > '1'
   and cast(date as numeric(38,0)) != sourceDate

  return
 end

 if @method = 'FIX'
 begin 

  declare @slinks slinks
  insert @slinks 
  select distinct slink
  from dbo.glDetailReports 
  where isnumeric(sourceDate+'.e0') = 1
   and sourceDate > '1'
   and cast(date as numeric(38,0)) != sourceDate


  update a set
   a.date = c.sourceDate
  from dbo.glDetail a
  join @slinks b on a.slink = b.slink
  join dbo.glDetailReports c on b.slink = c.slink

  return  
 end


end
