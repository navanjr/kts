create proc dbo.diagFix_paidLinkedToDeleted(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin

 if @method = 'GET'
 begin 
 
  select
   @guide = 'GET|SHOW',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(p.id),
   @message = 'Paid records linked to a deleted receipt' 
  from paid p
  left join Object o on cast(o.id as varchar) = dbo.slinkId(p.slink)
  where
   left(p.slink,1) = 'o'
   and o.typ < 0
 end

 if @method = 'SHOW'
 begin  

  select *
  from paid p
  left join Object o on cast(o.id as varchar) = dbo.slinkId(p.slink)
  where
   left(p.slink,1) = 'o'
   and o.typ < 0

 end

 if @method = 'FIX'
 begin  
 
  exec dbo.logit @@procId, 'Ok ill fix it...'

 end

end
