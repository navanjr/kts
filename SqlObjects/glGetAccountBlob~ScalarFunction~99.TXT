create function dbo.glGetAccountBlob(@glCode varchar(10)) returns varchar(500) as
begin

 declare @retvar varchar(500)

 select @retvar = '@glAccountCode=' + rtrim(key1) + ';'
  + '@glAccountDesc=' + rtrim(key2) + ';'
  + '@glAccountId=' + cast(id as varchar) + ';'
 from object where typ = 4701 and key3 = @glCode

 return isnull(@retvar,'')

end
