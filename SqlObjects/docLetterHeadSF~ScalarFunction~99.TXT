create function dbo.docLetterHeadSF(@id int) returns varchar(max)
   as
begin

	declare @addy varchar(2000)
 

 select @addy=
    case when len(officialName)>0 then officialName+ CHAR(13) + CHAR(10) else '' end 
  + case when len(county)>0 then county +' COUNTY TREASURER' + CHAR(13) + CHAR(10) else '' end 
  + case when len(add1)>0 then add1 + CHAR(13) + CHAR(10) else '' end 
  + case when len(add2)>0 then add2 + CHAR(13) + CHAR(10) else '' end 
  + case when len(city)>0 then city +', ' else '' end 
  + case when len(st)>0 then st +' ' else '' end 
  + case when len(zip)>0 then zip + CHAR(13) + CHAR(10) else '' end 
  + case when len(phone)>0 then 'PHONE ' + phone else '' end 
  from site

  
return @addy


end
