create function dbo.keyTemplates() returns @rt table(
 id int,
 template varchar(50),
 currentSize int,
 maxSize int,
 warnFlag varchar(50),
 lastEditDate varchar(10),
 lastEditTime varchar(10),
 color int,
 ord varchar(50)
)
begin

 insert @rt 
 select
  id * -1,
  template,
  len(cast(options as varchar(max))),
  0,
  '',
  cast(floor(lastedit) as varchar(10)),
  cast(floor((lastedit - FLOOR(lastedit)) * 10000) / (60 * 6.9445) as money),
  case when lastEdit > '  0' then 1 else 0 end,
  template
 from template
 where template > '  0'

 update a set a.maxSize = (select max(size) from dbo.temLog where template = a.template) from @rt a

 update @rt set warnFlag = 'suspect' where maxSize > currentSize

 return
end
