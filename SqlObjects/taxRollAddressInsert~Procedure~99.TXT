create procedure [dbo].[taxRollAddressInsert] @aId int
	as begin

insert into 
	taxRollDetail(
	    [adTaxId],
	    [parcelNumber],
	    [name],
	    [itemNumber],
	    [taxYear],
	    [Stamp],
	    [businessName],
	    [ADDRESS1],
		[ADDRESS2],
		[ADDRESS3],
		[CITY],
		[STATE],
		[ZIP1],
		[ZIP2],
		[ZIP3],
                [dType],
		[Comments]
	 	)	
select
	ID,
	FULLPIDNUMBER,
	OWNERNAME,
	ITEMNUMBER,
	REALTAXYEAR,
	GETDATE(),
	BUSINESSNAME,
	ADDRESS1,
	ADDRESS2,
	ADDRESS3,
	CITY,
	STATE,
	ZIP1,
	ZIP2,
	ZIP3,
        Recordtype,
	''

 from AdTax where id=@aId and (len(FULLPIDNUMBER)>0 or LEN(ownername)>0)

end
