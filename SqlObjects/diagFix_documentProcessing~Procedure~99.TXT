create proc dbo.diagFix_documentProcessing(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on

 if @method = 'GET'
 begin  

  select
   @class = 'Collections',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(a.id),  
   @message = 'missing bankrupt items in taxRollDetail'
  from documentProcessing a
  join taxrollDetail b on
   a.invoiceId = b.invoiceLinkId
   and a.documentName = b.documentName
   and a.processCode = b.Stamp

  return
 end

 if @method = 'SHOW'
 begin  

  select 
   a.*
  from documentProcessing a
  join taxrollDetail b on
   a.invoiceId = b.invoiceLinkId
   and a.documentName = b.documentName
   and a.processCode = b.Stamp

  return
 end

 if @method = 'FIX'
 begin 
   
  delete documentProcessing where id in
  (select 
   a.id
  from documentProcessing a
  join taxrollDetail b on
   a.invoiceId = b.invoiceLinkId
   and a.documentName = b.documentName
   and a.processCode = b.Stamp)
  
  return  
 end


end
