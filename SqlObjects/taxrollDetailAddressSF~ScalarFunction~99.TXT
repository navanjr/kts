create function dbo.taxrollDetailAddressSF(@taxrollId int) returns varchar(2000)
as
begin
declare @r varchar(2000)
 
select @r =     case when name>'  0' then rtrim(name)+' ' else '' end
		+case when businessName>'  0' then rtrim(businessName)+' ' else '' end
		+case when address1>'  0' then rtrim(address1)+' ' else '' end
		+case when address2>'  0' then rtrim(address2)+' ' else '' end
		+case when address3>'  0' then rtrim(address3)+' ' else '' end
		+case when city>'  0' then rtrim(city) else '' end
		+case when state>'  0' then ', '+rtrim(state) else '' end
		+case when zip1>'  0' then ' '+rtrim(zip1) else '' end
		+case when zip2>'  0' then '-'+rtrim(zip2) else '' end

           from taxrollDetail where id = @taxrollId
 set @r = isnull(@r,'')

return @r
end
