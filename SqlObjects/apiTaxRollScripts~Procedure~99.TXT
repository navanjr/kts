create procedure dbo.apiTaxRollScripts
as begin 
 SET DEADLOCK_PRIORITY LOW

 -- turn off the trigger on adtax
   alter table adtax disable trigger adTaxModifiedTR



  update AdTax set
         apiUpdated=''
       where 
        case when treamort > '  0' then treamort else cast(mortgageCode as varchar) end>'0'


   -- turn off the trigger on adtax
   alter table adtax enable trigger adTaxModifiedTR

end
