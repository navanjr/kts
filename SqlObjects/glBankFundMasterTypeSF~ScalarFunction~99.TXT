create function dbo.glBankFundMasterTypeSF(
 @accountType varchar(50)
) returns varchar(50) as
begin

-- this will return the masterAccountType for either a given accountType or a given accountId

 declare @retvar varchar(50)

 if isnumeric(@accountType) = 1
  select @accountType = accountType from dbo.glAccounts where accountId = @accountType

 select @retvar = masterAccountType from dbo.glBankFundTypes('') where accountType = @accountType

 return @retvar
end
