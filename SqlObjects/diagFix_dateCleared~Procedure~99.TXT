create proc dbo.diagFix_dateCleared(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin
 set nocount on

 if @method = 'GET'
 begin 
 
  select
   @class = 'BankStatement',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(g.id),
   @message = 'missing dateCleard on glDetail'
  from gldetail g
  join Object o on o.ID = g.bsId and o.Typ = 4780
  where isnull(g.bsId, 0) != 0
   and (isnull(g.dateCleared,0) = 0 or g.dateCleared not between o.a1 and o.a2)

 end

 if @method = 'SHOW'
 begin  

  select *
  from gldetail g
  join Object o on o.ID = g.bsId and o.Typ = 4780
  where isnull(g.bsId, 0) != 0
   and (isnull(g.dateCleared,0) = 0 or g.dateCleared not between o.a1 and o.a2)

 end

 if @method = 'FIX'
 begin

  update g set
   g.dateCleared = o.a2
  from gldetail g
  join Object o on o.ID = g.bsId and o.Typ = 4780
  where isnull(g.bsId, 0) != 0
     and (isnull(g.dateCleared,0) = 0 or g.dateCleared not between o.a1 and o.a2)
  
  exec dbo.logit @@procId, 'Just updated some invoices....', @@rowcount

 end

end
