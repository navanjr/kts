create function dbo.paymentTypes() returns @rt table(paymentType varchar(60),auditorReportGroup varchar(60),detail varchar(10)) as
begin
insert @rt
 select 'Miscellaneous','OTHER DISBURSEMENTS','Y'
  union
 select 'Treasurers Check','WARRANTS','N'
  union
 select 'Warrant','WARRANTS','N'
  union
 select 'Cancel Voucher','CALCELLED VOUCHERS','N'
  union
 select 'Official Voucher','VOUCHERS','N'
return
end
