create function dbo.paymentTypes() returns @rt table(paymentType varchar(60),auditorReportGroup varchar(60),detail varchar(10)) as
begin

declare @wt table(paymentType varchar(60),auditorReportGroup varchar(60),detail varchar(10))

insert @wt
 select 'Miscellaneous','OTHER DISBURSEMENTS','Y'
  union
 select 'Treasurers Check','WARRANTS','N'
  union
 select 'Warrant','WARRANTS','N'
  union
 select 'Cancel Voucher','CANCELLED VOUCHERS','N'
  union
 select 'Void Voucher','VOID VOUCHERS','N'
  union
 select 'Official Voucher','VOUCHERS','N'
  union
 select 'Trust Voucher','VOUCHERS','N'
  union
 select 'Cancel Miscellaneous','Cancelled Checks','N'

insert @rt 
 select key1, key2, key3 from object where typ=1515

insert @rt 
 select * from @wt where paymentType not in (select paymentType from @rt)

return
end
