create function dbo.docGetIndent(@doc varchar(max),@text varchar(100)) returns int
AS 
BEGIN
 declare @whereadd int, 
         @lineend varchar(2) = char(10),
 	 @indent int

 select @whereadd=charindex(@text,@doc)-1, @indent = 0
 if @whereadd > 0
  select @indent=charindex(@lineend,reverse(left(@doc,@whereadd)))-1
 return @indent
end
