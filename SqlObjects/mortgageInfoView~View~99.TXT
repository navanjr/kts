create view dbo.mortgageInfoView as
  select 'M'+cast(id as varchar) as id,receiptId,mortgagor,mortgagee,description,section,township,range,lender,creationdate,creationtime
    from mortgageInfo 
   union all 
  select 'O'+cast(id as varchar) as id,link1,key1,key2,d1,key3,a1,a2,a3,CreationDate,CreationTime 
     from Object where typ=4507
