create function dbo.receiptSLinks(@receiptId int) returns @rt table(
 slink varchar(15)
)
begin

 insert @rt select 'o' + cast(@receiptId as varchar)
 insert @rt select 't' + cast(invoiceId as varchar) from receiptLink with (NOLOCK) where receiptId = @receiptId 
 insert @rt select 'l' + cast(id as varchar) from receiptLink with (NOLOCK) where receiptId = @receiptId 
 insert @rt select 'j' + cast(id as varchar) from journalLink with (NOLOCK) where receiptId = @receiptId

 return
end
