create view dbo.receipt as
select 
 ID recId,
 key1 recNum,
 Key2 postDate,
 dbo.date1(key2) postdate1,
 b10 recDate,
 dbo.date1(b10) recdate1,
 key3 recType,
 A1 recFrom,
 A17 recStatus,
 A18 Deputy,
 A19 Official,
 dbo.SpellOut(c8) n2w,
 b13 as station,
 C8 Due,
 C9 Paid,
 D1 comments,
 'o'+cast(id as varchar(15)) slink
 from Object where Typ=4502
