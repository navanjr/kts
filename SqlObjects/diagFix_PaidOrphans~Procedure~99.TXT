create proc dbo.diagFix_paidOrphans(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin
 set nocount on

 if @method = 'GET'
 begin 
 
  select
   @class = 'receipts',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(*),
   @message = 'Orphaned paid records... deposit is missing'
  from paid
  where depositId != 0
   and depositId not in (select ID from Object where typ = 4513)

-- Remove this automatic repair as it messes up the data at Bryan County
--  exec dbo.diagnosticsAutoHelper @@procid, @blockAutoFix

  return
 end

 if @method = 'SHOW'
 begin  

  select *
  from paid
  where depositId != 0
   and depositId not in (select ID from Object where typ = 4513)

 end

 if @method = 'FIX'
 begin  
 
  exec dbo.logit @@procId, 'Ok ill fix it...'

  update paid set depositId = 0
  where depositId != 0
   and depositId not in (select ID from Object where typ = 4513)

 end

end
