create proc dbo.diagFix_paidOrphans(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin

 if @method = 'GET'
 begin 
 
  select
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(*),
   @message = 'Orphaned paid records... deposit is missing'
  from paid
  where depositId != 0
   and depositId not in (select ID from Object where typ = 4513)

 end

 if @method = 'SHOW'
 begin  

  select *
  from paid
  where depositId != 0
   and depositId not in (select ID from Object where typ = 4513)

 end

 if @method = 'FIX'
 begin  
 
  exec dbo.logit @@procId, 'Ok ill fix it...'

  update paid set depositId = 0
  where depositId != 0
   and depositId not in (select ID from Object where typ = 4513)

 end

end
