create procedure dbo.createGLCollectionDescriptions
 as 
begin

 set nocount on

 declare @wt table (
  name varchar(50),
  ord varchar(50)
 )

 insert @wt
  select 'Rcpts Other Than Ad Valorem Tax','02'
  union select 'Depository','03'
  union select 'Ad Valorem Tax','01'
  union select 'General','01'
  union select 'Deposits & Disbursements','05'

 delete @wt where name in (select key1 from object where typ = 4703)

 insert into object(typ,key1,key2) select 4703,name,ord from @wt


---clean up that mispelled descriptions
 update object set typ=-4703 where key1 in ('Receipts Other Than Ad Velorem Tax','Ad ValoremTax')
 update object set a4 = 'Rcpts Other Than Ad Valorem Tax' where typ=4701 and a4 = 'Receipts Other Than Ad Velorem Tax'
 update object set a4 = 'Ad Valorem Tax' where typ=4701 and a4 = 'Ad ValoremTax'

end
