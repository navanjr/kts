create proc dbo.diagFix_SQLObjects(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin
 set nocount on

 if @method = 'GET'
 begin 
 
  select 
   @class = 'General',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = COUNT(*),  
   @message = 'SQL Objects not OK'
  from dbo.sqlObjectCompare(0)
  where stat != 'OK' and skipFlag = 0  -- and name not like 'mike_%'

 end

 if @method = 'SHOW'
 begin  

  select * 
  from dbo.sqlObjectCompare(0)
  where stat != 'OK' and skipFlag = 0  -- and name not like 'mike_%'
  order by name

 end

 if @method = 'FIX'
 begin  

  exec dbo.keySQLObjectCreateAll
  return

 end

end
