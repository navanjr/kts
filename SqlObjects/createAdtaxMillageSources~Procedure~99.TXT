create procedure dbo.createAdtaxMillageSources
as 
begin
 declare @year varchar(4), @district varchar(50), @rate varchar(50), @taxname varchar(50), @sourcecode varchar(50), @newAccountId int 

 declare @pt table(realtaxyear varchar(4), ORGSCHOOLDISTRICTMAIN varchar(50), ORGSCHOOLDISTRICTTAXRATE varchar(50), taxname varchar(50), sourcecode varchar(50))

 insert @pt 
 select distinct cast([realtaxyear] as varchar(4)), [ORGSCHOOLDISTRICTMAIN], [ORGSCHOOLDISTRICTTAXRATE],m.a1,[ORGSCHOOLDISTRICTMAIN] + '_' +rtrim(m.a1)+ '_' + cast([realtaxyear] as varchar(4)) from adtax, object m 
 where m.typ=4012 and m.key1=[ORGSCHOOLDISTRICTMAIN] and m.key2=[ORGSCHOOLDISTRICTTAXRATE] and m.key3=[realtaxyear] and [realtaxyear]>0 and [ORGSCHOOLDISTRICTMAIN]>'0' and [ORGSCHOOLDISTRICTTAXRATE]>'0'
  
 delete from @pt where sourcecode in (select accountCode from glAccounts)


 while exists(select sourcecode from @pt)
 begin
  select top 1 @sourcecode=sourcecode from @pt
   
  exec dbo.glAccountVerification @sourcecode, 'SOURCE', 'CREATE', @accountId = @newAccountId OUTPUT
 
  delete from @pt where sourcecode=@sourcecode
 end
end
