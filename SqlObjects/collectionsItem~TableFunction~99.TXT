create function dbo.collectionsItem(
 @taxrollDetailId int
) returns @rt table(
 dtype varchar(50),
 parcel varchar(50),
 item numeric(7,1),
 keyCode varchar(50)
) as
begin

 insert @rt 
 select
  dtype,
  parcelNumber,
  itemNumber,
  keycode = case when dtype = 'RESALE' then parcelNumber else cast(itemNumber as varchar) end
 from taxrollDetail
 where id = @taxrollDetailId

 return
end
