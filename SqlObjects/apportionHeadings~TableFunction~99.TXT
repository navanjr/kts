create function dbo.apportionHeadings() returns @hd table(id int identity(1,1),
 ord int,
 heading varchar(60),
 accountCode varchar(60),
 accountDesc varchar(60),
 tableType varchar(60),
 toSchool varchar(1),
 toCity varchar(1),
 toVotech varchar(1))
as
begin
insert @hd 
 select 0,case when a13 > ' 0' then a13 else key1 end, key2, a18, a9, a3, a4, a5 from object where typ=4008 order by a9 desc, a3 desc, a4 desc, a5 desc, a18, key1
 
update @hd set ord=id where isnull(ord,0)=0
return
end
