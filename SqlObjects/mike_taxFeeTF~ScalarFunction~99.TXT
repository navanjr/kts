create function dbo.mike_taxFeeTF(
 @taxyear varchar(4)
) returns @rt table(
 code varchar(50),
 value varchar(50)
) as
begin

 insert @rt
 select 'mike_fee',
  dbo.padLeft(convert(varchar, count(*),1),'',9) + ' - ' + convert(varchar, sum(amount),1) from mike_fee where taxyear + itm_nbr in (select mikeFeeLink from dbo.invoiceAll where taxyear = @taxyear)


 return

end
