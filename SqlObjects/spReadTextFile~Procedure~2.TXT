create procedure dbo.spReadTextFile(
 @fullPath varchar(1000),
 @verbose varchar(5) = 'FALSE',
 @fileContents varchar(max) output
) as
begin

 declare 
  @sql nvarchar(max),
  @params nvarchar(max)

 select
  @sql = 'select @text_file=(select * from openrowset (bulk ''' + @fullPath + ''', SINGLE_CLOB) as X)',
  @params = '@text_file varchar(max) OUTPUT'  

 exec dbo.logit @@procId, '@fullPath', @fullPath

 exec sp_executesql
  @stmt = @sql,
  @params = @params,
  @text_file = @fileContents OUTPUT

if @verbose = 'TRUE'
 select @fileContents

end
