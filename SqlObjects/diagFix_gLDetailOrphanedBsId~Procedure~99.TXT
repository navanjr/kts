create proc dbo.diagFix_gLDetailOrphanedBsId(
 @mode varchar(5) = 'light',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on

 if @method = 'GET'
 begin  

  declare @rt table(bsId int,cnt int)


  insert @rt
  select
   bsId,
   count(*)
  from gldetail 
   where bsId not in (select id from object)
    and isnull(bsId,0)>1
   group by bsId


   select 
    @guide = 'GET|SHOW|FIX',
    @tally = sum(cnt),
    @code = case when count(*) > 0 then 1 else 0 end,  
    @message = 'glDetail orphaned bsId',
    @fixProc = 'exec dbo.diagFix_gLDetailOrphanedBsId @method = ''FIX'''
   from @rt

  return
 end

 if @method = 'SHOW'
 begin  
  select
   *
  from gldetail 
   where bsId not in (select id from object)
    and isnull(bsId,0)>1
 end

 if @method = 'FIX'
 begin  
  update gldetail 
  set bsId = 0
   where bsId not in (select id from object)
    and isnull(bsId,0)>1
 end
end
