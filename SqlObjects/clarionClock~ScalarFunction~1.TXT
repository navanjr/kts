create function [dbo].[clarionclock] (@x datetime) returns varchar(10) as  
begin 
 declare @y varchar(10) 
 set @y='' 
 if @x > '' 
  begin 
  select @y=(( 
 case when datediff(s,cast(@x as int),@x)>0 
  then datediff(s,cast(@x as int),@x) 
  else 86400+datediff(s,cast(@x as int),@x) 
 end) * 100)+1 
  end 
 return @y 
end
