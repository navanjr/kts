create function dbo.journalTypes() returns @rt table(journalType varchar(60),auditorReportGroup varchar(60),detail varchar(10)) as
begin
insert @rt
 select 'APPORTIONMENT','COLLECTIONS','N'
  union
 select 'APPORTIONMENT SPECIAL','COLLECTIONS','N'
  union
 select 'CORRECTION','JOURNAL','N'
  union
 select 'MANUAL','JOURNAL','N'
  union
 select 'Tax Refund','JOURNAL','N'
return
end
