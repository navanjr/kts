create function dbo.readStringSanitizer(
 @label varchar(50),
 @data varchar(max)
) returns varchar(max) as
begin
 
 if isnull(@label,'') > '  0'
  return '@' + @label + '=' + replace(replace(replace(@data,'@',''),'=',''),';','') + ';'

 return replace(replace(replace(@data,'@',''),'=',''),';','') 
end
