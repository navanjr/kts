create proc dbo.diagFix_adtaxMissingOrigin(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin
 set nocount on

 if @method = 'GET'
 begin 
 
  select 
   @class = 'Taxroll',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = COUNT(*),  
   @message = 'Origin Missing on AdTax Record'
  from adtax 
  where origin < ' 0'
   
 end

 if @method = 'SHOW'
 begin  

  select * 
  from adtax 
  where origin < ' 0'
 end

 if @method = 'FIX'
 begin  
  update adtax set origin = 'omitted'
  where isnull(certificatenumber,'NULL') = 'NULL' 
   and origin < ' 0'
   
  update adtax 
  set origin = 'Original Invoice'
  where origin < ' 0' 
  and isnull(certificatenumber,'NULL')<>'NULL'

  return

 end

end
