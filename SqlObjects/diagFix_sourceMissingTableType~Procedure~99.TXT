create proc dbo.diagFix_sourceMissingTableType(
 @mode varchar(5) = 'light',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin
 set nocount on

 declare @wt table(
  district varchar(50)
 )

 insert @wt
 select t.district
 from dbo.taxratedetail t
 group by t.district

 if @method = 'GET'
 begin 

  select 
   @class = 'tax',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = COUNT(*),  
   @message = 'Source accounts missing table type!'
  from glaccounts 
  where accounttype='SOURCE' 
   and aptabletype <>'M' 
   and apyear > ' 0' 
   and apdistrict > ' 0' 
   and aprate > ' 0'

  exec dbo.diagnosticsAutoHelper @@procid, @blockAutoFix

 end

 if @method = 'SHOW'
 begin  

  select * 
  from glaccounts 
  where accounttype='SOURCE' 
   and aptabletype <>'M' 
   and apyear > ' 0' 
   and apdistrict > ' 0' 
   and aprate > ' 0'

 end

 if @method = 'FIX'
 begin  

  update object set a9='M'
  where typ=4701 
   and a1='SOURCE' 
   and a9 <> 'M' 
   and a11 > ' 0' 
   and a12 > ' 0' 
   and a13 > ' 0'
  return

 end

end
