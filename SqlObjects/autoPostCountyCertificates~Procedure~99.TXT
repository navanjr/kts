create procedure dbo.autoPostCountyCertificates
as 
begin
declare @autoPostCountyCertificates varchar(5)
 select @autoPostCountyCertificates = settingvalue from settings where settingname='taxroll.autoPostCountyCertificates'

if @autoPostCountyCertificates = 'TRUE'
begin
 exec dbo.diagFix_countyCertificatesUpdate @method='fix'
end


end
