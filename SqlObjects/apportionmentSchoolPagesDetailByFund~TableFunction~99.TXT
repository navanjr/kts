	[accountCode] [varchar](60) NOT NULL default '',
	[accountDesc] [varchar](50) NOT NULL default '',

	[comment2] [varchar](50) NOT NULL default '',

	[journalDesc] [varchar](50) NOT NULL default '',
        [ord] [varchar](50) NOT NULL default '',
        [apportionContraCode] [varchar](50)

insert @det
select isnull(accountType,''),
	isnull(accountId,0),
	isnull(accountCode,''),
	isnull(accountDesc,''),
	isnull(comment,''),
	isnull(comment2,''),
	isnull(sourceCode,''),
	isnull(amount,0),
	isnull(journalType,''),
	isnull(sourceDesc,''),
	isnull(sourceSECA,''),
	isnull(journalDesc,''),
        '',
        apportionContraCode 
  from rpMonthEndDetail(@beginDate,@endDate, @includeUnapportioned) 
 where accountcode<>isnull(apportionContraCode,'NULL') and 
      accountType in (select accountType from dbo.glBankFundTypes('FUND'))
  and journalType = 'APPORTIONMENT'

if @includeUnapportioned='SPECIAL'
begin
insert @det
select isnull(accountType,''),
	isnull(accountId,0),
	isnull(accountCode,''),
	isnull(accountDesc,''),
	isnull(comment,''),
	isnull(comment2,''),
	isnull(sourceCode,''),
	isnull(amount,0),
	isnull(journalType,''),
	isnull(sourceDesc,''),
	isnull(sourceSECA,''),
	isnull(journalDesc,''),
        '',
        apportionContraCode
  from rpMonthEndDetail(@beginDate,@endDate, @includeUnapportioned) 
 where accountcode<>isnull(apportionContraCode,'') and 
      accountType in (select accountType from dbo.glBankFundTypes('FUND'))
  and journalType = 'APPORTIONMENT SPECIAL'
