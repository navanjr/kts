create function dbo.blobit(
 @pipedKeys varchar(max),
 @pipedValues varchar(max)
) returns varchar(max) as
begin

 declare @retMax varchar(max) = ''
 
 select
  @retMax = @retMax + '@' + dbo.stripChars(a.data,'64,61,59') + '=' + dbo.stripChars(b.data,'64,61,59') + ';' 
 from dbo.split(@pipedKeys,'|') a
 join dbo.split(@pipedValues,'|') b on a.id = b.id
 
 
 return @retMax

end
