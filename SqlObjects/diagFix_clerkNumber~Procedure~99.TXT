create proc dbo.diagFix_clerkNumber(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin

 if @method = 'GET'
 begin 
 
  select 
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = COUNT(*),  
   @message = 'Non standard officer/clerk numbers on payments'
  from Object
  where typ=4771 and rtrim(key3)<>rtrim(dbo.popro6(key3,6,0))
 end

 if @method = 'SHOW'
 begin  

  select * 
  from Object
  where typ=4771 and rtrim(key3)<>rtrim(dbo.popro6(key3,6,0))
 end

 if @method = 'FIX'
 begin  
  update object set key3=dbo.popro6(key3,6,0)   
    where typ=4771 and rtrim(key3)<>rtrim(dbo.popro6(key3,6,0))
  return

 end

end
