create function dbo.bankStatementDateCheck(@account varchar(50),@date int,@id int) 
 returns varchar(10)
 as
begin
 declare @rValue varchar(10) = ''
  select @rValue = 'EXISTS'  from object where typ = 4780 and Key1 = @account and @date between A1 and A2 and a1>'  0' and a2>'  0' and id <> @id
 
 return @rValue
end
