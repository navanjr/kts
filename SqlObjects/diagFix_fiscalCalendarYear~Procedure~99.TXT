create proc diagFix_fiscalCalendarYear(
 @mode varchar(5) = 'light',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on

 if @method = 'GET'
 begin  
  select
   @class = 'General',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(distinct id),  
   @message = 'Fiscal year missing on fiscal calendar.'
  from object 
  where typ=4700 
   and a18<' 0'

  exec dbo.diagnosticsAutoHelper @@procid, @blockAutoFix

  return
 end

 if @method = 'SHOW'
 begin  
  select * from object where typ=4700 and a18<' 0'

  return
 end

 if @method = 'FIX'
 begin 
  update object 
    set a18 = dbo.validateFiscalYear(left(key1,4))
   where typ=4700 and a18<' 0'
  return  
 end


end
