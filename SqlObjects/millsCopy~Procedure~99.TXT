create procedure dbo.millsCopy( 
 @typ int,
 @fromYear varchar(50),
 @toYear varchar(50)
)
as
begin

 insert object (typ,key1,key2,key3,a1,a2)
 select @typ,@toYear,key2,key3,a1,a2 from object where typ = @typ and key1 = @fromYear and @toYear+key2 not in (select key1+key2 from object where typ = @typ)
 
 return
end
