create procedure dbo.taxrollAddComment (@invoiceId int, @initials varchar(5)) as 

if @invoiceId<9000000
begin
insert into object (typ,key1,key2,key3,a1,a2,attributes)
select 4002,
       item,
       parcel,
       name,
       taxyear,
       dbo.clariondate(getdate()),
       @initials
 from invoices where id = @invoiceId
end
else
begin
insert into object (typ,key1,key2,key3,a1,a2,attributes)
select 4002,
       ITEMNUMBER,
       FULLPIDNUMBER,
       ownername,
       realtaxyear,
       dbo.clariondate(getdate()),
       @initials
 from adtax where id = @invoiceId-9000000
end
