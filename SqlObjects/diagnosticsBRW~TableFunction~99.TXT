create function dbo.diagnosticsBRW()
 returns @rt table(
  id int identity(1,1),
  method varchar(50),
  code int,
  message varchar(max),
  tally int,
  fixProc varchar(100)
) as 
begin

 declare 
  @methodUnit varchar(50),
  @code int = 0,
  @message varchar(max) = '',
  @tally int = 0

  exec dbo.diagFix_orphanedReceiptLinks
    @code = @code output,
    @tally = @tally output,
    @message = @message output

  insert @rt
   select @methodUnit, @code, @message, @tally ,'none' 

 return
end
