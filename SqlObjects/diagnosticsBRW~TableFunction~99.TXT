create function dbo.diagnosticsBRW(
 @mode varchar(10),
 @id int
)
 returns @rt table(
  id int,
  display varchar(max),
  method varchar(50),
  code int,
  tally int,
  message varchar(50),
  timeStamp datetime,
  ord varchar(50),
  colorFlag int
) as 
begin

 if @mode = 'INSPECT'
 begin
  insert @rt (id, method, code, tally, message, timeStamp)
  select
   id,
   replace(method,'diagFix_',''),
   code,
   tally,
   message,
   timeStamp
  from dbo.diagnosticResults
  where id = @id
 end

 if @mode in ('HEAVY','LIGHT')
 begin
  if not isnull(@id,0) > 0
   select @id = max(trialId) from dbo.diagnosticResults where mode = @mode

  insert @rt (id, method, code, tally, message, timeStamp)
  select
   id,
   replace(method,'diagFix_',''),
   code,
   tally,
   message,
   timeStamp
  from dbo.diagnosticResults
  where trialId = @id
  order by testId

  update @rt set 
   display = dbo.padRight(' ' + method, ' ',50) + case when tally > 0 then dbo.padLeft(convert(varchar, tally,1), ' ', 12) else '' end,
   ord = 'a' + method

  update @rt set 
   colorFlag = 1 
  where tally > 0

  insert @rt (id, display, ord, colorFlag)
  select 
   id,
   '   ~ ' + message,
   'a' + method + 'z',
   1 
  from @rt
  where colorFlag = 1

  insert @rt (display, ord)
  select
   upper(@mode) + ' - ' + cast(max(timeStamp) as varchar) + ' (' + dbo.formatAge(max(timeStamp),getdate()) + ' old)',
   'a' 
  from @rt 
 end

 return
end
