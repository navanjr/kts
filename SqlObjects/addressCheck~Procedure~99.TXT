create procedure dbo.addressCheck @id int
	as 
begin
	declare	@exists int,
		@taxRollId int
				

	select @taxRollId = TAXROLLID from invoices with (NOLOCK) where id = @id
	
	select @exists=COUNT(id) from taxRollDetail with (NOLOCK) where adTaxId=@taxRollId and dtype='a'
 
 if @exists+0=0
	begin
		exec dbo.taxRollAddressInsert @taxRollId
	end
 end
