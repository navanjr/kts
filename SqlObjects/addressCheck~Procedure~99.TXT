create procedure dbo.addressCheck @id int
	as 
begin
	declare	@exists int,
			@taxRollId int
				
	set @Id=146917

	select @taxRollId = TAXROLLID from invoices
	
	select @exists=COUNT(id) from taxRollDetail where adTaxId=@taxRollId
 
 if @exists+0=0
	begin
		exec dbo.taxRollAddressInsert @taxRollId
	end
 end
