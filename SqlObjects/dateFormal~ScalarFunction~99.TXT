create  function [dbo].[dateFormal] (@x int) returns varchar(50) as 
begin
 declare	@y varchar(50),
		@date varchar(12) = rtrim(convert(char,cast(cast(@x-36163 as int) as datetime),112))

 set @y=''
 if isnull(@x,'')<>'' and isnumeric(@x)=1 and @x>0
 begin
  set @y = upper(datename(M,@date))+' '+cast(datepart(D,@date) as varchar)+', '+cast(datepart(YEAR,@date) as varchar)
 end
 return @y
end
