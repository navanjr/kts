create procedure dbo.glAccountVerification (
-- this routine should be the only place we insert accural glaccounts 
-- TODO: this needs to be enhanced to handle pilot, ada, population, etc...
-- so we expect an accural accountCode (with the _XX) and accountType and if its not found we create it
 @accountCode varchar(50),
 @accountType varchar(50),
 @mode varchar(50),
 @tableType varchar(50) = '',
 @pilotTable varchar(50) = '',
 @year varchar(50) = '',
 @district varchar(50) = '',
 @rate varchar(50) = '',
 @populationType varchar(50) = '',
 @accountId int = null OUTPUT
) as
begin
 
-- first we check for this account if we have it we get the id and bail
 if exists(select * from glAccounts where accountCode = upper(@accountCode) and accountType = upper(@accountType))
 begin
  select @accountId = accountId from glAccounts where accountCode = upper(@accountCode) and accountType = upper(@accountType)
  return
 end

 declare
  @accountDesc varchar(50),
  @typeDesc varchar(50),
  @reportCode varchar(50)

  select 
   @typeDesc = key2,
   @reportCode = key3
  from object where typ = 4702 and key1 = upper(@accountType)

  declare @accountTypes table(accountType varchar(50), accountDesc varchar(50))
  insert @accountTypes select 'SOURCE', replace(@accountCode,'_',' ')
  insert @accountTypes select 'PAYABLE', replace(replace(@accountCode,'_AP',' Payable'),'_',' ')
  insert @accountTypes select 'RECEIVABLE', replace(replace(@accountCode,'_AR',' Receivable'),'_',' ')
  insert @accountTypes select 'ACCRUAL', replace(replace(@accountCode,'_AC',' Accrual'),'_',' ')

  if upper(@mode) = 'CREATE' and upper(@accountType) in (select accountType from @accountTypes)
  begin
   select @accountDesc = accountDesc from @accountTypes where accountType = @accountType
   select @accountId = @@identity
   insert object (typ,key1,key2,a1,a2,a3,a9,a10,a11,a12,a13,a14,a16) 
    select 4701,upper(@accountCode),@accountDesc,upper(@accountType),@typeDesc,@reportCode,@tableType,@pilotTable,@year,@district,@rate,@populationType,'KTS' 
  end

  return

end
