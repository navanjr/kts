create procedure dbo.journalEntryCRUD(
 @mode int,
 @id int = null,
 @postDate int = null,
 @description varchar(50) = '',
 @jeType varchar(50) = '',
 @comment varchar(max) = '',
 @conversionNote varchar(16) = '',
 @verbose varchar(5) = 'TRUE',
 @resultCode int = 0 output,
 @resultMessage varchar(50) = '' output,
 @newId int = null output
) as 
begin
 set nocount on

 declare
  @slink varchar(15) = 'o' + cast(@id as varchar)

 select @resultCode = 0, @resultMessage = 'Success'

 if @postDate is null
  select @postDate = dbo.clarionDate(getDate())

 exec dbo.logit @@procid, '@mode', @mode

 -- Delete
 if @mode = 3
 begin
  begin transaction
  exec dbo.glDeleteObjectSource @id, @deletePosted = 'True'
  delete glDetail where slink in (select 'j' + cast(id as varchar) from journalLink where jeId = @id)
  delete journalLink where jeId = @id
  -- remove the bsId link on glDetail
  update glDetail set bsId = null where bsId = @id
  commit transaction
 end

 -- Update
 if @mode = 2
 begin
  declare 
   @b14_absAmount money,
   @b15_balance money
  select 
   @b14_absAmount = absAmount,
   @b15_balance = balance
  from glSummary(@id,'o')
  update object set
   b14 = cast(@b14_absAmount as varchar),
   b15 = cast(@b15_balance as varchar)
  where typ = 4512 and id = @id
 end
  
 -- Create
 if @mode = 0
 begin
  -- bail if missing description or Journal Type
  if @description < '  0' or @jeType < '  0'
  begin 
   if @verbose = 'TRUE'
    select @resultCode = 1, @resultMessage = 'this method requires Description and Journal Type'
   return
  end
  
  declare
   @jeNumber varchar(50)
 
  exec dbo.nextObjectAutoNumber 4512, null, @newNumber = @jeNumber output

  declare @newIdWt newIdWt
  insert object (typ,key1,key2,key3,a18,k2,e1,olink5)
  output inserted.id into @newIdWt
  select 4512, @jeNumber, @postDate, @description, @jeType, @postDate, @comment, @conversionNote

  exec dbo.logit @@procid, 'inserted J/E  @conversionNote', @conversionNote

  select @newId = max(id) from @newIdWt

 end

 return
end
