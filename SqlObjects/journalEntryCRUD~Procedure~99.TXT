create procedure dbo.journalEntryCRUD(
 @mode int,
 @id int = null,
 @postDate int = null,
 @description varchar(50) = '',
 @jeType varchar(50) = '',
 @resultCode int = 0 output,
 @resultMessage varchar(50) = '' output,
 @newId int = null output
) as 
begin

 select @resultCode = 0, @resultMessage = 'Success'

 if @postDate is null
  select @postDate = dbo.clarionDate(getDate())

 -- Delete
 if @mode = 3
 begin
  exec dbo.glDeleteObjectSource @id
  delete journalLink where jeId = @id
 end

 -- Update
 if @mode = 2
 begin
  declare 
   @b14_absAmount money,
   @b15_balance money
  select 
   @b14_absAmount = absAmount,
   @b15_balance = balance
  from glSummary(@id,'o')
  update object set
   b14 = cast(@b14_absAmount as varchar),
   b15 = cast(@b15_balance as varchar)
  where typ = 4512 and id = @id
 end
  
 -- Create
 if @mode = 0
 begin
  -- bail if missing description
  if @description < '  0' or @jeType < '  0'
  begin 
   select @resultCode = 1, @resultMessage = 'this method requires Description and Journal Type'
   return
  end
  
  declare
   @padLength int,
   @suffixLength int,
   @keyCodeOptions varchar(50),
   @jeNumber varchar(50)
 
  -- TODO: put this in a nextNumber() function
  select @keyCodeOptions = dbo.readKeyCode(4512,'\0002')
  select 
   @padLength = cast(substring(@keyCodeOptions,1,2) as int),
   @suffixLength = cast(substring(@keyCodeOptions,3,2) as int)

  -- TODO: autonumber with Suffix will not be compatible with this... fix it later
  select @jeNumber =  cast(cast(max(case when isnumeric(key1) = 1 then key1 else '0' end) as int) + 1 as varchar) from object where typ = 4512
  select @jeNumber = dbo.padLeft(@jeNumber,'0',@padLength)

  insert object (typ,key1,key2,key3,a18)
  select 4512, @jeNumber, @postDate, @description, @jeType

  set @newId = @@identity

 end

 return
end
