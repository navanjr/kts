create function dbo.taxrollSearchCommentBRW(
 @selectedId int
) returns @rt table(
 Id int,
 noteDate nvarchar(12),
 actionDate nvarchar(12),
 comment text
)
begin

 declare
  @name varchar(50),
  @item numeric (7,1),
  @parcel varchar(50),
  @taxYear numeric (5,0)

 select 
  @name = name,
  @item = item,
  @parcel = parcel,
  @taxYear = taxyear
 from invoices where id = @selectedId
 
  insert @rt
  select id,a2,a3,e1
  from dbo.object
  where typ=4002 
  and (case when isnumeric(key1)=1 and key1>'0' then cast(key1 as numeric (7,1)) else 0.0 end=@item or len(key1)<4)
  and (key2=@parcel or len(key2)<1) 
  and (key3=left(@name,len(key3)) or len(key3)<1) 
  and (case when isnumeric(a1)=1 then cast(a1 as numeric (5,0)) else 0 end =@taxyear or len(a1)<2)

 return
end
