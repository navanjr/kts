create function dbo.officialBankAccountNo(@officialId int) returns varchar(50)
as 
begin
declare	@r varchar(50),
		@accountCode varchar(50)
		
select @accountCode = A4 from Object  where ID = @officialId
 
select @r  = A6 from Object  where TYP = 4701 and Key1=@accountCode
 
 return @r
end
