create view dbo.taxrollAuditDetailPenaltyAdjust
as
select i.id as invoiceId,sum(isnull(a.amount,0)) as PenaltyAmount
from dbo.gldetailreports a, invoices si, invoices i
 where left(a.slink,1)='j'
  and a.invoiceid = si.id
  and si.invoiceid = i.id
  and si.typ in ('P')
  and accounttype = 'ACCRUED RECEIVABLE'
group by i.id
