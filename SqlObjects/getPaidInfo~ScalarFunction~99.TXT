create function dbo.getPaidInfo(@id int) returns varchar(500) as 
begin
 declare @retvar varchar(500)
 
 select @retvar = '@paycode=' + paycode + ';' 
 + '@amount=' + cast(amount as varchar) + ';' 
 + '@checkNumber=' + checkno + ';' 
 + '@drawnon=' + drawnon + ';' 
 + '@location=' + location + ';' 
 from paid where id = @id

 return @retvar
end
