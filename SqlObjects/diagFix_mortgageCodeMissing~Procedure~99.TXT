create proc diagFix_mortgageCodeMissing(
 @mode varchar(5) = 'light',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on

 if @method = 'GET'
 begin  

  select
   @class = 'General',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(distinct treamort),  
   @message = 'Mortgage Codes Missing'
  from adtax 
   where treamort > ' 0' 
     and treamort not in (select key1 from object where typ=4030)

  exec dbo.diagnosticsAutoHelper @@procid, @blockAutoFix

  return
 end

 if @method = 'SHOW'
 begin  

  select distinct treamort
  from adtax 
   where treamort > ' 0' 
     and treamort not in (select key1 from object where typ=4030)

  return
 end

 if @method = 'FIX'
 begin 

  insert object (typ,key1,key3,a8)
  select distinct 4030, treamort, 'New '+dbo.date1(dbo.clariondate(getdate())), 'Created by Diagnostic'
  from adtax 
   where treamort > ' 0' 
     and treamort not in (select key1 from object where typ=4030)

  return  
 end


end
