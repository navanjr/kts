create function dbo.taxRateMillageNew(
 @year varchar(50),
 @district varchar(50),
 @rateName varchar(50)
) returns @rt table(
 setId int,
 millageName varchar(50),
 millageAmount varchar(50)
) as

begin

 insert @rt (millageName) select key1 from object where typ = 4008

 update @rt set
  millageAmount = (select top 1 a2 from object where typ = 4012 and key1 = @district and key2 = @rateName and key1 = @year and a1 = millageName) 

 return
end
