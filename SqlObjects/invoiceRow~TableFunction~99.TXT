create function dbo.invoiceRow(@invoiceId int) returns @rt table(
 id int identity(1,1),
 ord varchar(50),
 fieldName varchar(50),
 fieldValue varchar(50)
)
begin

 declare @invoice table(
  taxYear varchar(4),
  taxrollId int,
  parcel varchar(50),
  name varchar(50)
 )


 insert @rt select 'a','TaxYear',taxYear from invoices with (NOLOCK) where id = @invoiceId
 insert @rt select 'b','TaxrollId',cast(taxrollId as varchar) from invoices with (NOLOCK) where id = @invoiceId
 insert @rt select 'c','Parcel',parcel from invoices with (NOLOCK) where id = @invoiceId
 insert @rt select 'd','Name',name from invoices with (NOLOCK) where id = @invoiceId
 insert @rt select 'e','Status',status from invoices with (NOLOCK) where id = @invoiceId
 insert @rt select 'f','Date',dbo.date1(postDate) from invoices with (NOLOCK) where id = @invoiceId

 return

end
