create function dbo.isOfficialBank(@accountCode varchar(60)) returns varchar(5) as
begin
declare @isOfficialBank varchar(5)

 set @isOfficialBank = 'FALSE'

 if exists(select * from object where typ=4601 and a4=@accountCode)
 begin
  set @isOfficialBank = 'TRUE'
 end

return @isOfficialBank
end
