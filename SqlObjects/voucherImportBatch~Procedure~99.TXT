create procedure dbo.voucherImportBatch( @batchId int ) as 
begin

  declare
   @idToken int,
   @voucherDate int,
   @voucherNumber varchar(50),
   @payee varchar(50),
   @amount varchar(50),
   @debitAcct varchar(50),
   @creditAcct varchar(50) 

  declare @wt table (
   id int,
   voucherDate int,
   voucherNumber varchar(50),
   payee varchar(50),
   amount money,
   debitAcct varchar(50),
   creditAcct varchar(50)
  )
  insert @wt select 
   id,
   cast(voucherDate as int),
   voucherNumber,
   payee,
   cast(amount as money),
   debitAcct,
   creditAcct
  from dbo.voucherImportBRW(@batchId)
 
  while exists(select * from @wt)
  begin
    select top 1 @idToken = id, 
     @voucherDate = voucherDate,
     @voucherNumber = voucherNumber,
     @payee = payee,
     @amount = amount,
     @debitAcct = debitAcct,
     @creditAcct = creditAcct
    from @wt order by id

    exec dbo.voucherCRUD 0,  
     @voucherDate = @voucherDate,
     @voucherNumber = @voucherNumber,
     @payee = @payee,
     @amount = @amount,
     @debitAcct = @debitAcct,
     @creditAcct = @creditAcct

    delete @wt where id = @idToken
  end

 return
end
