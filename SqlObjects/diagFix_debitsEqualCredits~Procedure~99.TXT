create proc dbo.diagFix_debitsEqualCredits(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin

 if @method = 'GET'
 begin 
 
  select 
   @guide = 'GET|SHOW',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = COUNT(*),  
   @message = 'slinks out of balance'
  from gldetail
  group by slink
  having SUM(amount) !=0

 end

 if @method = 'SHOW'
 begin  

  select slink 
  from gldetail
  group by slink
  having SUM(amount) !=0

 end

 if @method = 'FIX'
 begin  
 
  return

 end

end
