create proc dbo.diagFix_debitsEqualCredits(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin
 set nocount on

 if @method = 'GET'
 begin 
 
  select 
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = COUNT(*)
  from gldetail
  group by slink
  having SUM(amount) !=0

  select 
   @class = 'General',
   @guide = 'GET|SHOW',
   @message = 'slinks out of balance',
   @code = coalesce(@code,0),
   @tally = coalesce(@tally,0)

 end

 if @method = 'SHOW'
 begin  

  select slink 
  from gldetail
  group by slink
  having SUM(amount) !=0

 end

 if @method = 'FIX'
 begin  
 
  return

 end

end
