create proc dbo.diagFix_orphanedGLDetail(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on

 if @method = 'GET'
 begin  

  if @mode != 'heavy'
  begin
   select
    @class = 'General',
    @guide = 'GET|SHOW|FIX',
    @code = 0, 
    @message = 'SKIPPED - orphaned glDetail - Heavy proc'
   return
  end

  declare @rt table(typ varchar(1),cnt int)


  insert @rt
  select
   left(slink,1),
   count(*)
  from gldetail 
   where dbo.slinkType(slink)='o' 
    and dbo.slinkId(slink) not in (select id from Object where typ > 0)
   group by left(slink,1)

  insert @rt
  select
   left(slink,1),
   count(*)
  from gldetail 
   where dbo.slinkType(slink)='l' 
     and dbo.slinkId(slink) not in (select id from receiptlink)
   group by left(slink,1)

  insert @rt
  select
   left(slink,1),
   count(*)
  from gldetail 
   where dbo.slinkType(slink)='j' 
     and dbo.slinkId(slink) not in (select id from journallink)
   group by left(slink,1)

  insert @rt
  select
   left(slink,1),
   count(*)
  from gldetail 
   where dbo.slinkType(slink)='t' 
     and dbo.slinkId(slink) not in (select id from invoices)
   group by left(slink,1)

   select 
    @guide = 'GET|SHOW|FIX',
    @tally = sum(cnt),
    @code = case when count(*) > 0 then 1 else 0 end,  
    @message = 'orphaned glDetail',
    @fixProc = 'exec dbo.diagFix_orphanedGLDetail @method = ''FIX'''
   from @rt

  return
 end

 if @method = 'SHOW'
 begin  
  select
   *
  from gldetail 
   where dbo.slinkType(slink)='o' 
    and dbo.slinkId(slink) not in (select id from Object)

  select
   *
  from gldetail 
   where dbo.slinkType(slink)='l' 
     and dbo.slinkId(slink) not in (select id from receiptlink)

  select
  *
  from gldetail 
   where dbo.slinkType(slink)='j' 
     and dbo.slinkId(slink) not in (select id from journallink)

  select
   *
  from gldetail 
   where dbo.slinkType(slink)='t' 
     and dbo.slinkId(slink) not in (select id from invoices)



 end

 if @method = 'FIX'
 begin  

  delete
  from gldetail 
   where dbo.slinkType(slink)='o' 
    and dbo.slinkId(slink) not in (select id from Object)

  delete
  from gldetail 
   where dbo.slinkType(slink)='l' 
     and dbo.slinkId(slink) not in (select id from receiptlink)

  delete
  from gldetail 
   where dbo.slinkType(slink)='j' 
     and dbo.slinkId(slink) not in (select id from journallink)

  delete
  from gldetail 
   where dbo.slinkType(slink)='t' 
     and dbo.slinkId(slink) not in (select id from invoices)
  
 end


end
