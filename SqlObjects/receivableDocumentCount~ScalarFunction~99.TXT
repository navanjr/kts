create function dbo.receivableDocumentCount(
 @documentName varchar(50),
 @date int,
 @days int
) returns int as
begin

 declare @tally int

 select @tally = count(*) 
 from taxrollDetail
 where dtype = 'print'
  and documentName = @documentName
  and documentDate between @date and @date + @days

 return @tally
 
end
