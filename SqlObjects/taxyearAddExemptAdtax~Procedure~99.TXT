create procedure dbo.taxyearAddExemptAdtax @taxYear varchar(4) 
as 
begin

 insert into invoices (TAXROLLID,TYP,PARCEL,ITEM,TAXYEAR,NAME,BUSINESSNAME,POSTDATE,STATUS)
	select	
        id,
	RECORDTYPE,
	FULLPIDNUMBER,
	ITEMNUMBER,
	REALTAXYEAR,
	OWNERNAME,
	BUSINESSNAME,
	'',
	'Exempt'

      from AdTax where 
	id not in (select taxrollId from invoices where typ<>'s') 
	and REALTAXYEAR = @taxYear 
	and OWNERNAME>'  0'
        and TOTALDUE=0
        
end
