create procedure dbo.paymentGetNextvoucherNumber(
 @voucherId int,
 @series varchar(50)
) as
begin

 declare
  @retval nvarchar(max),
  @log varchar(max)
   
 exec dbo.nextObjectAutoNumber 4771, @series, @newNumber = @retval output

 select @log = '@code=0;@nextVoucherNumber=' + @retval + ';'

 exec dbo.logit @@procId, @log

 select @log

end
