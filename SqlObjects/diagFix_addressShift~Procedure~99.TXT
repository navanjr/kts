create proc dbo.diagFix_addressShift(
 @mode varchar(5) = 'light',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on

 if @method = 'GET'
 begin  

  select
   @class = 'Taxroll',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(a.id),  
   @message = 'Shifted addresses'
  from adtax a
   where ADDRESS1<' 0' and (ADDRESS2 > ' 0' or ADDRESS3 > ' 0') and realtaxyear > 2013


  return
 end

 if @method = 'SHOW'
 begin  

  select 
   a.realtaxyear,
   a.itemnumber,
   a.fullpidnumber,
   a.ownername,
   a.address1,
   a.address2,
   a.address3
  from adtax a
   where ADDRESS1<' 0' and (ADDRESS2 > ' 0' or ADDRESS3 > ' 0') and realtaxyear > 2013

  return
 end

 if @method = 'FIX'
 begin  

  update adtax 
   set address1 = address2,
    address2 = address3,
    address3 = ''
   from adtax a
   where ADDRESS1<' 0' and (ADDRESS2 > ' 0' or ADDRESS3 > ' 0') and realtaxyear > 2013

  update adtax 
   set address1 = address2,
    address2 = address3,
    address3 = ''
   from adtax a
   where ADDRESS1<' 0' and (ADDRESS2 > ' 0' or ADDRESS3 > ' 0') and realtaxyear > 2013

  update adtax 
   set address2 = address3,
    address3 = ''
   from adtax a
   where ADDRESS2<' 0' and ADDRESS3 > ' 0' and realtaxyear > 2013


  return  
 end


end
