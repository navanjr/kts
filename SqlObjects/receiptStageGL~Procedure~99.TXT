create procedure dbo.receiptStageGL (@sourceId int) as
begin
 
 declare @accountsWorking table(
  accountId int,
  amount money
 )
 declare @accounts table(
  accountId int,
  amount money
 )
 
 declare 
  @code int,
  @contraId int,
  @arAccountId int,
  @tokenId int,
  @tokenAmount money
 
 select 
  @code = code,
  @contraId = contraId ,
  @arAccountId = arAccountId
 from dbo.receiptCheck(@sourceId,null)

 -- credit Source and Debit AR
 insert @accountsWorking
  select b.accountId, sum(a.amount)*-1 
  from receiptDetail a, dbo.glAccounts b where a.receiptId = @sourceId and a.sourceCode = b.accountCode group by b.accountId
 select @tokenAmount = abs(sum(amount)) from @accountsWorking
 insert @accountsWorking select @arAccountId, @tokenAmount

 if exists(select * from paid where sourceId = @sourceId)
 begin
  declare @ap table(accountId int, amount money, prorate decimal(19,18), payAmount decimal(19,10))
  declare @paidTotal money

  -- credit AR
  select @paidTotal = sum(amount)*-1 from paid where sourceId = @sourceId
  insert @accountsWorking select @arAccountId, @paidTotal

  -- debit suspense accounts
  insert @accountsWorking
  select b.link1, sum(a.amount)
  from paid a, object b 
  where a.paycode = b.key1 and b.typ = 4505 and a.sourceId = @sourceId group by b.link1

  -- debit tresurers receipt fund and credit fund payables
  insert @accountsWorking select @contraId, @paidTotal
  insert @ap
   select b.accountId,sum(a.amount)*-1, 0, 0
   from receiptDetail a, dbo.glAccounts b
   where a.receiptId = @sourceId and rtrim(a.fundCode)+'_AP' = b.accountCode group by b.accountId

  update @ap set prorate = amount / cast(@tokenAmount as decimal(19,10))
  update @ap set payAmount = round(@paidTotal * prorate,2)
 end 
 
 insert @accountsWorking select accountId, payAmount from @ap

 insert @accounts select accountId, sum(isnull(amount,0)) from @accountsWorking group by accountId having sum(isnull(amount,0)) != 0

 -- remove 
 delete glDetailStage where sourceId = @sourceId and accountId not in (select accountId from @accounts)

 -- update
 update a set a.amount = b.amount
 from glDetailStage a, @accounts b
 where a.accountId = b.accountId
  and a.sourceId = @sourceId
 
 delete @accounts where accountId in (select accountId from glDetailStage where sourceId = @sourceId)

 -- insert
 while (select count(*) from @accounts) > 0
 begin
  select top 1 @tokenId = accountId, @tokenAmount = amount from @accounts
  insert gldetailstage (sourceId,accountId,amount,contraId) select @sourceId,@tokenId,@tokenAmount,@contraId
  delete @accounts where accountId = @tokenId
 end
 
 update a set
  a.accountCode = b.accountCode,
  a.accountDesc = b.accountDesc
 from glDetailStage a, glAccounts b
 where a.accountId = b.accountId
  and a.sourceId = @sourceId 

end
