create function dbo.glBankFundTypes(@type varchar(50)) returns @rt table(accountType varchar(50), ord varchar(50))
as
begin

 if @type = 'BANK' 
  begin
   insert @rt
    select 'BANK', 'B1' 
      union select 'SUSPENSE', 'B2'
      union select 'INVESTMENT', 'B2'
  end

 if @type = 'FUND' 
  begin
   insert @rt
    select 'FUND' , 'F5'
      union select 'OFFICIAL', 'F4'
      union select 'VOTECH', 'F2'
      union select 'SCHOOL', 'F1'
      union select 'CITY', 'F3'
  end

 if @type = 'BOTH' 
  begin
   insert @rt
    select 'FUND' , 'F5'
      union select 'OFFICIAL', 'F4'
      union select 'VOTECH', 'F2'
      union select 'SCHOOL', 'F1'
      union select 'CITY', 'F3'
      union select 'BANK', 'B1'
      union select 'SUSPENSE', 'B2'
  end

 return
end
