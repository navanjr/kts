create function dbo.glBankFundTypes(@type varchar(50)) returns @rt table(accountType varchar(50))
as
begin

 if @type = 'BANK' 
  begin
   insert @rt
    select 'BANK' 
      union select 'SUSPENSE'
  end

 if @type = 'FUND' 
  begin
   insert @rt
    select 'FUND' 
      union select 'OFFICIAL'
      union select 'VOTECH'
      union select 'SCHOOL'
      union select 'CITY'
  end

 if @type = 'BOTH' 
  begin
   insert @rt
    select 'FUND' 
      union select 'OFFICIAL'
      union select 'VOTECH'
      union select 'SCHOOL'
      union select 'CITY'
      union select 'BANK' 
      union select 'SUSPENSE'
  end

 return
end
