create proc dbo.diagFix_MissingPropLoc(
 @mode varchar(5) = 'light',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin
 set nocount on

 if @method = 'GET'
 begin 
 
  select 
   @class = 'adtax',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = COUNT(*),  
   @message = 'Missing Property locations that we can find for you.'
  from adtax
  where
   len(proploc)<1
      and (PHYSICALSTREETNUMBER > ' 0'
     or PHYSICALSTREET > ' 0'
     or PHYSICALTOWN > ' 0'
     or PHYSICALSTREETDIRECTION > ' 0')

  exec dbo.diagnosticsAutoHelper @@procid, @blockAutoFix

 
 end

 if @method = 'SHOW'
 begin  

  select * from adtax
  where
   len(proploc)<1
      and (PHYSICALSTREETNUMBER > ' 0'
     or PHYSICALSTREET > ' 0'
     or PHYSICALTOWN > ' 0'
     or PHYSICALSTREETDIRECTION > ' 0')

 end

 if @method = 'FIX'
 begin  

  update adtax set proploc = ltrim(left(rtrim(PHYSICALSTREETNUMBER)+' '+rtrim(PHYSICALSTREETDIRECTION)+' '+rtrim(PHYSICALSTREET)+' '+rtrim(PHYSICALTOWN),50)), modified = getdate()
  where
   len(proploc)<1
      and (PHYSICALSTREETNUMBER > ' 0'
     or PHYSICALSTREET > ' 0'
     or PHYSICALTOWN > ' 0'
     or PHYSICALSTREETDIRECTION > ' 0')
 end

end
