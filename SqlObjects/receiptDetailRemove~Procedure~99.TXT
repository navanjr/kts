create procedure dbo.receiptDetailRemove(@receiptDetailId int, @sourceId int) as
begin

 declare 
  @code int,
  @message varchar(100),
  @receiptType varchar(50),
  @suspenseAccountId int,
  @contraId int,
  @fpId int
  
 select 
  @code = code,
  @message = message,
  @receiptType = receiptType,
  @suspenseAccountId = suspenseAccountId,
  @contraId =contraId,
  @fpId = fpId
 from dbo.receiptCheck(@sourceId,null)


 -- bail if we dont have enough data
 if @code = 1
 begin
  select '@code=1;@message=' + @message + ';'
  return
 end


 begin transaction
  delete receiptDetail where id = @receiptDetailId
  
/*
  exec dbo.receiptStagePayments @sourceId
  
  if (select sum(amount) from dbo.glDetailStage where sourceId = @sourceId) <> 0
   rollback transaction
  else
*/
   commit transaction

 select '@code=0;@message=Payment removed OK.;'
 return

end
