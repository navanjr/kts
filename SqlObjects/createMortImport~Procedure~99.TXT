create proc dbo.createMortImport as
begin

 if exists(select * from dbo.sysobjects where id = object_id(N'[dbo].[MortImport]') and OBJECTPROPERTY(id, N'IsTable') = 1)
 begin
  exec dbo.logit @@procid, 'dropping table MortImport...'
  drop table dbo.MortImport
 end

 exec dbo.logit @@procid, 'creating table MortImport...'

CREATE TABLE [dbo].[MortImport](
        [AUTONUMBERKEY] nvarchar(100),
        [MORTGAGECODE] int,
	[NAME] [nvarchar](30) NULL,
	[ADDRESS1] [nvarchar](30) NULL,
	[ADDRESS2] [nvarchar](30) NULL,
	[CITY] [nvarchar](30) NULL,
	[STATE] [nvarchar](30) NULL,
	[COUNTRY] [nvarchar](30) NULL,
	[ZIP1] [nvarchar](30) NULL,
	[ZIP2] [nvarchar](30) NULL,
	[ZIP3] [nvarchar](30) NULL,
        [PHONEVOICE] nvarchar(30) NULL,
        [PHONEFAX] nvarchar(30) NULL,
        [EMAILADDRESS] nvarchar(30) NULL,
        [CHANGEDATE] int NULL,
        [CHANGEINITIALS] nvarchar(30) NULL,
	[COMMENTS] [nvarchar](50) NULL
) ON [PRIMARY]

 
end
