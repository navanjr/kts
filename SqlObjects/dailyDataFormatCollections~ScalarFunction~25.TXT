create function dbo.dailyDataFormatCollections(
 @description varchar(50),
 @minControlNumber varchar(50),
 @maxControlNumber varchar(50),
 @rowCount int,
 @amount money,
 @descriptionLength int
) returns varchar(100) as 
begin

 declare
  @retval varchar(100),
  @rowCountString varchar(15) = '',
  @amountString varchar(15) = ''

 if @rowCount is not null
  set @rowCountString = cast(@rowCount as varchar)

 if @amount is not null
  set @amountString = convert(varchar,@amount,1)

 select @retval = 
  '   ' + dbo.padRight(@description, ' ', @descriptionLength)
  + case when @minControlNumber is null then replicate(' ',13) else dbo.padLeft(@minControlNumber, 0, 5) + case when @maxControlNumber>'  0' then ' - ' + dbo.padLeft(@maxControlNumber, 0, 5) else '        ' end end
  + ' ' + dbo.padLeft(@rowCountString, ' ', 4)
  + dbo.padLeft(@amountString,' ',16)

 return @retval
end
