create function dbo.keyColor(
 @red int,
 @green int,
 @blue int
) returns int as
begin
 declare @returnColor int

 select @returnColor = 
  convert(varbinary(1),@blue) 
  + convert(varbinary(1),@green) 
  + convert(varbinary(1),@red)

 return @returnColor 
end
