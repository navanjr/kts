create function dbo.taxCorrectionsSF(@adTaxValue varchar(200),@taxrollId int,@crLabel varchar(50)) returns varchar(50)
as
begin

declare @x varchar(200)


select top 1 
      @x = A2 
    from Object 
   where link1 = @taxrollId 
         and TYP=4005 
         and key1 = @crLabel 
     order by ID desc

if @x is null
 
 set @x = rtrim(@adTaxValue)

return @x

end
