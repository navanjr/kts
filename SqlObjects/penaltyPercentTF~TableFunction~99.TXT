Create Function penaltyPercentTF (
 @invoiceId int,
 @paydate varchar(10)
) returns @rt table(
 interestpercent decimal(21,9),
 months int,
 days int,
 rate int,
 calcMethod varchar(1) 
)as 
begin
 declare @grace int,
        @months int,
        @days int,
        @rate int,
        @begindate varchar (10),
        @enddate varchar (10),
        @typ varchar (1),
        @calcMethod varchar (1),
        @interestpercent decimal (21,9),
        @interestDate int
        
        
        
 select @typ = [typ], @interestDate = interestDate from invoices where id=@invoiceId
 select top 1
  @calcMethod = a19, 
  @grace = cast(a20 as int),
  @rate  = cast(b1 as int)
 from object
 where typ=40
 
 if @interestDate > 0
 begin
  select @begindate = dbo.date1(@interestDate)
 end
 else 
 begin
  if @typ in ('P','R','B','I')
  begin
   declare @totalDue money, @balanceDue money
  select 
       @totalDue = sum(case when accountType = 'SOURCE' then amount * -1 else 0 end),
       @balanceDue = sum(case when accountType = 'RECEIVABLE' then amount else 0 end)
       from dbo.invoiceGLTotalTF(@invoiceId)
         where status='Posted'   

--   select @totalDue = sum(amount) from dbo.invoiceTotalTF(@invoiceId) where accountType = 'RECEIVABLE' and amount > 0.00
--   select @balanceDue = sum(amount) from dbo.invoiceTotalTF(@invoiceId) where accountType = 'RECEIVABLE'
   
   if @balanceDue-.01 <= @totalDue/2
   begin
    select @begindate = '03/31/' + cast(taxyear+1 as varchar) from invoices where id = @invoiceId
   end
   else
   begin
    select @begindate = '12/31/' + cast(taxyear as varchar) from invoices where id = @invoiceId
   end
  end 
  else
  begin
   select @begindate = dbo.date1(postdate) from invoices where id = @invoiceId
  end
 end

 if datepart(m,@paydate) = 12
 begin
  select @enddate = case when day(@paydate) <= 16 then @paydate else '01/01/' + cast(year(@paydate) + 1 as varchar) end
 end
 else
 begin
  select @enddate = case when day(@paydate) <= 16 then @paydate else cast(datepart(m,@paydate) + 1 as varchar) + '/01/' + cast(year(@paydate) as varchar) end
 end

 set @days = datediff(d, @begindate, @enddate)

 set @months = datediff(m, @begindate, @paydate) 
 if day(@paydate) < @grace
 begin
  set @months = @months - 1
 end

 if @calcMethod = 'M'
 begin
  set @interestpercent = @months * @rate / 12.0
 end
 else
 begin
  set @interestpercent = @days * @rate / 365.0
 end 

 if @interestpercent > 100
  set @interestpercent = 100

 if @interestpercent < 0
  set @interestpercent = 0

 insert @rt
 select
  @interestpercent,
  @months,
  @days,
  @rate,
  @calcMethod

 return

end
