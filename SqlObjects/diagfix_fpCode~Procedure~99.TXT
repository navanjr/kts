create proc dbo.diagFix_fpCode(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on

 if @method = 'GET'
 begin  

  select
   @class = 'General',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(g.id),  
   @message = 'glDetail linked to the wrong/missing fiscal Calendar'
  from gldetail g
   join dbo.fiscalCalendar f on f.fpCode = g.fpCode
  where g.date not between f.startDate and f.endDate

  exec dbo.diagnosticsAutoHelper @@procid, @blockAutoFix

  return
 end

 if @method = 'SHOW'
 begin  

  select 
   g.slink,
   g.fpCode, f.fpCode,
   g.fpid, f.fpid,
   f.startDate, f.endDate, g.date,
   dbo.date1(g.date)
  from gldetail g
   join dbo.fiscalCalendar f on f.fpCode = g.fpCode
  where g.date not between f.startDate and f.endDate

  return
 end

 if @method = 'FIX'
 begin 

  declare @slinks table(slink varchar(15))
  insert @slinks
  select distinct
   g.slink
  from gldetail g
   join dbo.fiscalCalendar f on f.fpCode = g.fpCode
  where g.date not between f.startDate and f.endDate
  
  update g set
   g.fpCode = f.fpCode,
   g.fpId = f.fpId
  from gldetail g
   left join dbo.fiscalCalendar f on g.date between f.startDate and f.endDate
  where slink in (select slink from @slinks)
   and f.fpCode is not null

  return  
 end


end
