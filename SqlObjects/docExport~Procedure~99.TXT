create procedure docExport(
 @documentId int = 0,
 @method varchar(10)
) as
begin
 set nocount on

 exec dbo.logit @@procid, '@documentId', @documentId, '@method', @method

 declare
  @documentName varchar(50) = '',
  @path varchar(1000),
  @fullPath varchar(1000),
  @data varchar(max) = ''
 
 select @documentName = key1 from object where id = @documentId
 select @path = dbo.settingsF('git.gitpath','')
 select @fullPath = @path + '\documents\' + @documentName + '.txt'

exec dbo.logit @@procid, '@fullPath', @fullPath

 if @method = 'export'
 begin
  
  select @data = e1
  from object where id = @documentId
  if len(@data) > 0
    exec dbo.spOverwriteTextFile @fileName = @fullPath, @text = @data
   else
    select '@code=1;@message=Nothing to Export!;' 
  return
 end

end
