create procedure dbo.taxYearImport as

begin
 
 declare @x varchar(max)
 declare @wt table (taxYear varchar(10),invoices varchar(20))

insert into @wt
	select 
		REALTAXYEAR,
		COUNT(id) 
	from adtax 
		group by REALTAXYEAR 
	order by REALTAXYEAR

delete from @wt where taxYear in (select key1 from Object where Typ=4000)
insert into Object (typ,Key1,A1)
	select 4000,taxYear, invoices from @wt


end
