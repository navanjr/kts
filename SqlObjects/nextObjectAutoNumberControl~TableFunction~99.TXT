create function dbo.nextObjectAutoNumberControl(@typ int,@seriesfield varchar(50)) 
returns varchar(50)
as
begin
declare @seriesString varchar(50)

if @typ = 4771 and @seriesfield in ('Official Voucher','Trust Voucher')
 begin
  select @seriesString = 'Official Voucher'
 end
else
 begin
  select @seriesString = @seriesfield
 end

return @seriesString
end
