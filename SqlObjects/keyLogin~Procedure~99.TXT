create proc dbo.keyLogin (
 @initials varchar(5),
 @station varchar(50)
) as 
begin

 exec dbo.logit @@procid, 'Logging in @initials', @initials, '@station', @station

 insert login (ini,station) select @initials, @station

 declare 
  @code int = 0,
  @message varchar(max) = ''

 exec dbo.passwordPolicyCheck @initials, @code = @code output, @message = @message output

 select '@code=' + cast(@code as varchar) + ';@message=' + @message + ';'

end
