create function dbo.auditorReportBankReconciliation(@begindate varchar(15),@enddate varchar(15)) 
returns @rt table(
 auditorReportGroup varchar(50),
 auditReportGroup varchar(60),
 amount money,
 reportOrder varchar(60)
)

as 
begin

insert @rt 
select 'Cash In Banks','Cash In Banks',sum(case when isnumeric(a3)=1 then cast(a3 as money) else 0 end),'10' from object where typ=4780 and a2<=@enddate and a2>=@begindate

insert @rt 
select 'Cash On Hand','Cash On Hand',sum(amount),'15' from gldetailreports where sourceDate<=@endDate and accountType='CASH'

insert @rt
select '',*,'20' from dbo.auditorReportOSDetail(@beginDate,@endDate)

return
end
