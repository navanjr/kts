create function dbo.docCollectionDescResale(@interestDate varchar(50)) 
returns varchar(max)
as 
begin


declare @ret varchar(max) = '',
		@key1 varchar(10),
		@a17 varchar(10)
		
select top 1 @key1 = key1, @a17 = a17 from Object where TYP = 4026 order by ID desc



select  @ret = @ret 
			+ CHAR(13) + CHAR(10) 
			+ dbo.docCollectionDesc(invoiceIdMain,@interestDate,1) 
		from dbo.collectionsBRW('RESALE',0,@key1,@a17,0) where rowType = 0 order by ord




return @ret
end
