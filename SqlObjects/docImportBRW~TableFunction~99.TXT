create function dbo.docImportBRW(@x int)
 returns @rt table
(name VARCHAR(2000), 
 path VARCHAR(2000), 
  id int)
as 
begin
declare
  @path varchar(1000)
  
  
 select @path = dbo.readString('@gitPath=', b13) + '\documents\' from object where typ = 0 and link1 = -1

 insert @rt
  select name,path,id from dbo.dirRead(@path)

 return
end
