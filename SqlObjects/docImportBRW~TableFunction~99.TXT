create function dbo.docImportBRW(@x int)
 returns @rt table
(name VARCHAR(2000), 
 path VARCHAR(2000), 
  id int)
as 
begin
declare
  @path varchar(1000)
  
  
 select @path =  path + '\documents\' 
 from dbo.paths() where name = 'gitRepo'

 insert @rt
  select name,path,id from dbo.dirRead(@path)

 return
end
