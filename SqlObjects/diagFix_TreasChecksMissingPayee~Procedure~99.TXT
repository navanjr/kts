create proc dbo.diagFix_TreasChecksMissingPayee(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on

 if @method = 'GET'
 begin  

  select
   @class = 'Payments',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(id),  
   @message = 'Treasures Checks missing Warrant Payee'
  from dbo.payments 
  where paymentType = 'Treasurers Check' and warrantPayee = '' and id in (select warrantlinkId from payments)

  return
 end

 if @method = 'SHOW'
 begin  

  select 
   *
  from dbo.payments
  where paymentType = 'Treasurers Check' and warrantPayee = ''

  return
 end

 if @method = 'FIX'
 begin  
  update tc
   set tc.c5 = w.payee 
  from object tc, payments w
  where w.warrantLinkId = tc.id

  return  
 end


end
