create function dbo.journalSearch(@search varchar(15)) returns @rt table(id int)
 begin

 insert @rt
    select id from object where typ=4512 and (key1+key3+a18+dbo.date1(key2) like '%'+@search+'%' or e1 like '%'+@search+'%')

 if isnumeric(@search)=1
  begin
   insert @rt
     select cast(substring(slink,2,14) as int) from gldetail where abs(amount)=cast(@search as money)
   
  end


 return
 end
