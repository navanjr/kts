create function dbo.glGetAllSlinks(@slink varchar(15)) returns @rt table(
 slink varchar(15)
)
begin
  
 insert @rt select @slink

 -- if the source slink is an object records it could be a receipt or JE, better check for receipt and journal links
 if left(@slink,1) = 'o'
  insert @rt select 'l'+cast(id as varchar) from receiptLink where receiptId = cast(substring(@slink,2,14) as int)
  insert @rt select 'j'+cast(id as varchar) from journalLink where jeId = cast(substring(@slink,2,14) as int)

 return
end
