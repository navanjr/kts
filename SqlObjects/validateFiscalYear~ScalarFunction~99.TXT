create function dbo.validateFiscalYear(@inputYear varchar(20)) returns varchar(10)
as
begin
 declare
  @fiscalYear varchar(10),
  @input4 varchar(4)


 if len(@inputYear) = 4 and isnumeric(@inputYear) = 1
  select @input4 = @inputYear
 else
  select @input4 = '20' + right(@inputYear,2)

select @fiscalyear = cast(cast(@input4 as int) - 1 as varchar(4)) + '-' + @input4 where @input4 > ' 0' and isnumeric(@input4) = 1

return @fiscalYear
end
