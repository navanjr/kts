create function dbo.validateFiscalYear(@inputYear varchar(20)) returns varchar(10)
as
begin
declare @fiscalYear varchar(10),
        @input2 varchar(2)

select @input2=right(@inputYear,2)
select @fiscalyear = cast(cast('20'+@input2 as int)-1 as varchar(4))+'-'+'20'+@input2 where @input2 > ' 0' and isnumeric(@input2)=1

return @fiscalYear
end
