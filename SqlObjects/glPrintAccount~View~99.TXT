create view dbo.glPrintAccount 
 as 
select *,
 case 
  when sourcetype = '4502' then rtrim(sourcekey3)+' Receipt - Control #'+rtrim(sourcekey1) + ' - Received Of: ' + rtrim(sourcea1)
  when sourcetype = '4512' then rtrim(sourcea18)+' Journal #'+rtrim(sourcekey1)+' - Purpose: '+rtrim(sourcekey3)
  when sourcetype = '4513' then 'Deposit #' + rtrim(sourcekey1)
  when sourcetype = '4522' then 'Official Deposit #' + rtrim(sourcekey1)
  when sourcetype = '4771' then rtrim(sourcea18)+' - Treasurer #' + rtrim(sourcekey1) + ' - Vendor: ' + rtrim(sourcea2)
  else sourcetype
   end as printdescription
from dbo.gldetailreports a
