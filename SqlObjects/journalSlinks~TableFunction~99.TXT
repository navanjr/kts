create function dbo.journalSlinks(@jeId int) returns @rt table(
 slink varchar(15)
)
begin

 insert @rt select 'o' + cast(@jeid as varchar)
 insert @rt select 't' + cast(invoiceId as varchar) from journalLink where jeid = @jeid 
 insert @rt select 'j' + cast(id as varchar) from journalLink where jeid = @jeid

 return
end
