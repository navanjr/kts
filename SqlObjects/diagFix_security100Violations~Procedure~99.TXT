create proc dbo.diagFix_security100Violations(
 @mode varchar(5) = 'light',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on

 if @method = 'GET'
 begin  

  select
   @class = 'Security',
   @guide = 'GET|SHOW',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(id),  
   @message = 'Non KellPro Users with 100 security Clearance'
  from users 
  where Options not like '%@tim;%' and SecurityLevel=100

  return
 end

 if @method = 'SHOW'
 begin  

  select 
   FullName,
   securitylevel 
  from users 
  where Options not like '%@tim;%' and SecurityLevel=100

  return
 end



end
