create function dbo.taxrollSAImportTF(
 @taxyear varchar(4)
) returns @rt table(
 code varchar(50),
 value varchar(50)
) as
begin

 insert @rt
 select 'mike_taxroll',
  convert(varchar, count(*),1) + ' - ' + convert(varchar, cast(sum(tax_due) as money),1)
 from mike_taxroll where taxyear = @taxyear
  and tax_due != 0


 return

end
