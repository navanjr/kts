create procedure dbo.pilotFundReceivableVerification (@year varchar(4), @district varchar(50), @rate varchar(50), @pilottable varchar(50)) as
begin
 
 declare @fundCode varchar(50)

  if @district<='0' and @rate>'0' and @year>'0'
  begin
   set @fundCode = @rate+'_'+@year+'_AC'
   exec dbo.glAccountVerification @fundCode, 'ACCRUAL', 'CREATE', @tableType = 'S', @year = @year, @rate = @rate
  end

  if @district>'0' and @rate>'0' and @year>'0'
  begin
   set @fundCode = @district+'_'+@rate+'_'+@year+'_AC'
   exec dbo.glAccountVerification @fundCode, 'ACCRUAL', 'CREATE', @tableType = 'M', @year = @year, @district = @district, @rate = @rate
  end

  if @pilottable>'0' and @year>'0'
  begin
   set @fundCode = @pilottable+'_'+@year+'_AC'
   exec dbo.glAccountVerification @fundCode, 'ACCRUAL', 'CREATE', @tableType = 'L', @pilotTable = @pilotTable, @year = @year
  end

end
