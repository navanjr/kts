create procedure dbo.populationFundReceivableVerification (@year varchar(4), @poptype varchar(1)) as
begin
   
   declare
    @typename varchar(3)
   
   if len(@poptype)>0
   begin
    set @typename=@poptype+'_'
   end
   else
   begin
    set @typename=''
   end

 declare @fundCode varchar(50) = 'POPULATION_'+@typename+@year+'_AR'
 exec dbo.glAccountVerification @fundCode, 'RECEIVABLE', 'CREATE', @tableType = 'A', @year = @year, @populationType = @popType

end
