create view dbo.glBankAccounts as
-- this View will reveal all bank accounts that are properly setup for use with the daily reports

select 
 a.id as accountId,
 a.key1 as accountCode,
 a.key2 as accountDesc,
 a.a1 as accountType,
 a.a4 as collectionDescription,
 b.key2 as reportOrder
from object a, object b
where
 a.a4 = b.key1
 and a.typ = 4701
 and a.a1 = 'Bank'
 and a.a4 > '  0'
 and b.typ = 4703
