create function dbo.docDoIndent(@rawText varchar(max), @vname varchar(100), @value varchar(max)) returns varchar(max)
AS
BEGIN
 declare @indent int,
         @spaces varchar(100), 
		 @le varchar(2)

 select @indent = dbo.docGetIndent(@rawText,@vname),
        @spaces = replicate(' ',@indent),
		@le = char(13)+char(10)

 if @indent > 0
 begin
  set @value = replace(@value,@le,@le+@spaces) 
 end
 return @value
END
