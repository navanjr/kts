create proc dbo.diagFix_gldetailACRACRproblem(
 @mode varchar(5) = 'light',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on


 if @method = 'GET'
 begin  

  select
   @class = 'General',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(g.id),  
   @message = 'glDetail displaying ACRACR bug.'
  from gldetail g
  where RIGHT(g.accountcode,8)='_ACR_ACR'

  exec dbo.diagnosticsAutoHelper @@procid, @blockAutoFix

  return
 end

 if @method = 'SHOW'
 begin  

  select 
   g.slink,
   g.accountdesc
  from gldetail g
  where RIGHT(g.accountcode,8)='_ACR_ACR'

  return
 end

 if @method = 'FIX'
 begin 
  update gldetail set accountcode = left(accountCode,LEN(accountcode)-4) where RIGHT(accountcode,8)='_ACR_ACR'
  update object set typ=-4701 where typ=4701 and right(key1,8)='_ACR_ACR'
  update g set accountid=a.accountid, accountdesc=a.accountdesc from gldetail g, glaccounts a where g.accountcode=a.accountcode

  return  
 end


end
