create function dbo.receiptPayments(@recId int) returns @rt table(recTyp varchar(50),amount money)
as
begin


insert @rt
 
  select paycode+case when checkno>'  0' then '  -  '+checkno else '' end,amount 
       from paid with (NOLOCK) where 
        left(slink,1)='o' and dbo.slinkId(slink)=@recId
        and charindex('Returned',checkno)=0


 return
end
