create proc dbo.diagFix_splitproplocationAAMaster(
 @mode varchar(5) = 'light',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin
 set nocount on

 if @method = 'GET'
 begin 
 
  select 
   @class = 'AdTax',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = COUNT(*),  
   @message = 'Prperty Locations that can be split into specific fields'
  from adtax a, aamaster aa
  where a.origin <> 'mike_taxroll'
   and a.realtaxyear = aa.taxyear
   and a.fullpidnumber=aa.FULLPIDNUMBER 
   and a.FULLPIDNUMBER>''
   and len(rtrim(a.proploc))>2  and a.proploc > ' 0'
   and a.PHYSICALSTREETNUMBER < ' 0'
   and a.PHYSICALSTREET < ' 0'
   and a.PHYSICALTOWN < ' 0'

 end

 if @method = 'SHOW'
 begin  

  select * 
  from adtax a, aamaster aa
  where a.origin <> 'mike_taxroll'
   and a.realtaxyear = aa.taxyear
   and a.fullpidnumber=aa.FULLPIDNUMBER 
   and a.FULLPIDNUMBER>''
   and len(rtrim(a.proploc))>2  and a.proploc > ' 0'
   and a.PHYSICALSTREETNUMBER < ' 0'
   and a.PHYSICALSTREET < ' 0'
   and a.PHYSICALTOWN < ' 0'
 end

 if @method = 'FIX'
 begin  
  update a 
   set PHYSICALSTREETNUMBER = aa.PHYSICALSTREETNUMBER,
    PHYSICALSTREET = aa.PHYSICALSTREET,
    PHYSICALTOWN = aa.PHYSICALTOWN
  from adtax a, aamaster aa
  where a.origin <> 'mike_taxroll'
   and a.realtaxyear = aa.taxyear
   and a.fullpidnumber=aa.FULLPIDNUMBER 
   and a.FULLPIDNUMBER>''
   and len(rtrim(a.proploc))>2  and a.proploc > ' 0'
   and a.PHYSICALSTREETNUMBER < ' 0'
   and a.PHYSICALSTREET < ' 0'
   and a.PHYSICALTOWN < ' 0'
  return

 end

end
