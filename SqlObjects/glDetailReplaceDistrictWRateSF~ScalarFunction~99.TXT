create function dbo.glDetailReplaceDistrictWRateSF(@sourceCode varchar(50),@description varchar(50)) returns varchar(50)
	as 
begin

 declare	@r varchar(250),
			@district varchar(250),
			@rate varchar(250)		
select 
	 @district = case when charindex('_',@sourceCode)>0 then dbo.splitf(@sourceCode,'_',1) else '' end,
	 @rate = dbo.splitf(@sourceCode,'_',2)
 
select 
	@r = REPLACE(@description,@district,@rate) 

 return @r 
end
