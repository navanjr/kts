create function dbo.clarionKeyLastEditDateTime(
 @theDate datetime
) returns varchar(13) as
begin

 declare @retval varchar(13)
 
 select @retval = 
  right(
   '000000000000' + CAST(dbo.clarionDate(@theDate) as varchar)
   + '.' + 
   cast(ceiling((cast(dbo.clarionClock(@theDate) as int) / 8640000.0) * 10000) as varchar)
  ,12)

 return @retval

end
