create proc dbo.diagFix_miscAssessed(
 @mode varchar(5) = 'light',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as
begin
 set nocount on

 if @method = 'GET'
 begin  

  select
   @class = 'Taxroll',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(a.id),  
   @message = 'Missing or incorrect Net Assessed Value'
  from adtax a
   where GROSSASSESSED <> MiscAssessed+LandAssessed+ImprovedAssessed+MFGHOMEASSESSED
     and GROSSASSESSED > 0
     and realtaxyear > 2012


  return
 end

 if @method = 'SHOW'
 begin  

  select 
   a.realtaxyear,
   a.itemnumber,
   a.fullpidnumber,
   a.ownername
  from adtax a
   where GROSSASSESSED <> MiscAssessed+LandAssessed+ImprovedAssessed+MFGHOMEASSESSED
     and GROSSASSESSED > 0
     and realtaxyear > 2012

  return
 end

 if @method = 'FIX'
 begin  

  update adtax set MiscAssessed = GROSSASSESSED - (LandAssessed+ImprovedAssessed+MFGHOMEASSESSED)
   from adtax a
   where GROSSASSESSED <> MiscAssessed+LandAssessed+ImprovedAssessed+MFGHOMEASSESSED
    and GROSSASSESSED > 0
    and realtaxyear > 2012


  return  
 end


end
