create proc dbo.diagFix_stagedRecordsOnPosted(
 @mode varchar(5) = 'heavy',
 @method varchar(5) = 'GET',
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin

 if @method = 'GET'
 begin 

  select
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = count(distinct slink),
   @message = 'TRUST receipts with the Wrong/Old suspense glDetail'
  from gldetailStage where slink in (select slink from glDetail)

 end

 if @method = 'SHOW'
 begin  

  select * from gldetailStage where slink in (select slink from glDetail)

 end

 if @method = 'FIX'
 begin

  delete gldetailStage where slink in (select slink from glDetail)
  exec dbo.logit @@procId, 'Just deleted me some glDetailStage....', @@rowcount

 end

end
