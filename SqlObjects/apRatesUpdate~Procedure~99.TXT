create procedure dbo.apRatesUpdate ( @objTyp int )
as 
begin

-- turn off the trigger on object
  alter table object disable trigger keySqlObjectWriteToRepoTR


 insert object (typ, key1)
 select @objTyp, rateName from dbo.apRatesBRW(0) where rateName not in (select key1 from object where typ = @objTyp)
 if @objTyp=4008 update object set a3='1' where typ=4008 and left(key1,3)='SCH'

-- turn the trigger back on
  alter table object enable trigger keySqlObjectWriteToRepoTR


end
