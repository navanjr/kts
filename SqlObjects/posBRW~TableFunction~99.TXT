create function dbo.posBRW() returns @rt table(
 id int default 0,
 display varchar(500),
 depositDate int,
 amount money,
 accountId int,
 blob varchar(500) default '',
 ord varchar(50)
) as 
begin

 insert @rt (display,depositDate,amount,accountId,ord)
 select
  '   ' + receiptType,
  depositDate, 
  sum(amount),
  min(accountId),
  cast(depositDate as varchar) + 'b' 
 from dbo.posTF()
 group by id, depositDate, receiptType

 insert @rt (display,accountId,ord)
 select
  dbo.date1(depositDate),
  1, 
  cast(depositDate as varchar) + 'a' 
 from @rt
 where depositDate is not null
 group by depositDate

 return
end
