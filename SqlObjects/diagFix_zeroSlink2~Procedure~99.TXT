create proc dbo.diagFix_zeroSlink2(
 @mode varchar(5) = 'light',
 @method varchar(5) = 'GET',
 @blockAutoFix varchar(5) = 'FALSE',
 @class varchar(50) = null output,
 @code int = 0 output,
 @message varchar(100) = null output,
 @tally int = 0 output,
 @fixProc varchar(100) = null output,
 @guide varchar(100) = null output
) as 
begin
 set nocount on

 if @method = 'GET'
 begin 
 
  select 
   @class = 'gldetail',
   @guide = 'GET|SHOW|FIX',
   @code = case when count(*) > 0 then 1 else 0 end,
   @tally = COUNT(*),  
   @message = 'GLDetail with bad slink2'
  from gldetail
  where slink2='o0'
 end

 if @method = 'SHOW'
 begin  

  select * 
   from gldetail
   where slink2='o0'
 end

 if @method = 'FIX'
 begin  
  update gldetail set slink2=null
   where slink2='o0'
  return

 end

end
