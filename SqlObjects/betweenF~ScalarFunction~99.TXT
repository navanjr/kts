create function dbo.betweenF(
 @value varchar(50),
 @this varchar(50),
 @that varchar(50),
 @then varchar(50),
 @else varchar(50)
) returns varchar(50) as
begin

 declare @retval varchar(50)
 
 if @value between @this and @that
  set @retval = @then

 else
  set @retval = @else

 return @retval
end
