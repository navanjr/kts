create proc dbo.createTaxNoteImport as
begin

 if exists(select * from dbo.sysobjects where id = object_id(N'[dbo].[TaxNoteImport]') and OBJECTPROPERTY(id, N'IsTable') = 1)
 begin
  exec dbo.logit @@procid, 'dropping table TaxNoteImport...'
  drop table dbo.TaxNoteImport
 end

 exec dbo.logit @@procid, 'creating table TaxNoteImport...'

CREATE TABLE [dbo].[TaxNoteImport](
	[ITEMNUMBER] [decimal](7, 1) NULL,
	[USERINITIALS] [nvarchar](30) NULL,
	[NOTEDATE] [nvarchar](20) NULL,
	[NOTETIME] [nvarchar](20) NULL,
	[NOTECODE] [nvarchar](20) NULL,
	[ACTIONDATE] [nvarchar](20) NULL,
	[NOTE] varchar(max) NULL
) ON [PRIMARY]


 if not exists(select * from dbo.sysobjects where id = object_id(N'[dbo].[TaxNoteAll]') and OBJECTPROPERTY(id, N'IsTable') = 1)
 begin
  exec dbo.logit @@procid, 'creating table TaxNoteAll...'

  CREATE TABLE [dbo].[TaxNoteAll]([ID] [int] IDENTITY(1,1) NOT NULL,
        [REALTAXYEAR] int,
	[ITEMNUMBER] [decimal](7, 1) NULL,
	[USERINITIALS] [nvarchar](30) NULL,
	[NOTEDATE] [nvarchar](20) NULL,
	[NOTETIME] [nvarchar](20) NULL,
	[NOTECODE] [nvarchar](20) NULL,
	[ACTIONDATE] [nvarchar](20) NULL,
	[NOTE] varchar(max) NULL
  ) ON [PRIMARY]


 end

 
end
